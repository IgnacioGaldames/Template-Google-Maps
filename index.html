Multiples Marcadores

                    <script type="text/javascript">
                    var locations4 = [
                   ['Bondi Beach', -33.890542, 151.274856, 4],
                   ['Coogee Beach', -33.923036, 151.259052, 5],
                   ['Cronulla Beach', -34.028249, 151.157507, 3],
                   ['Manly Beach', -33.80010128657071, 151.28747820854187, 2],
                   ['Maroubra Beach', -33.950198, 151.259302, 1]
                 ];
                 var origin4 = new google.maps.LatLng(locations4[0][1], locations4[0][2]);

                 var map4 = new google.maps.Map(document.getElementById('map_multiple'), {
                   zoom: 10,
                   center: origin4,
                   mapTypeId: google.maps.MapTypeId.ROADMAP
                 });

                 var infowindow4 = new google.maps.InfoWindow();

                 var marker4, i;

                 for (i = 0; i < locations4.length; i++) {
                   var position4 = new google.maps.LatLng(locations4[i][1], locations4[i][2]);

                   marker4 = new google.maps.Marker({
                     position: position4,
                     map: map4
                   });

                   google.maps.event.addListener(marker4, 'click', (function(marker4, i) {
                     return function() {
                       infowindow4.setContent(locations4[i][0]);
                       infowindow4.open(map4, marker4);
                     }
                   })(marker4, i));
                 }
                    </script>
